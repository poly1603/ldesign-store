```typescript

class LRUCache {
  private cache = new Map()  // O(1) 查找
  private head/tail          // O(1) 添加/删除
  
  get(key) { /* O(1) */ }
  set(key, value) { /* O(1) */ }
}
```

### 3. 优先级桶订阅

```typescript
// 避免每次排序
class SubscriptionManager {
  private priorityBuckets = new Map<number, Set<Callback>>()
  
  subscribe(event, callback, priority) {
    // O(1) 添加到对应优先级桶
  }
  
  notify(event, data) {
    // O(k) 按优先级顺序执行
  }
}
```
